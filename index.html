Generate Number C program

// generate bunch of random phone numbers

#include
#include

#define trunkPrefix 0

int operators(int array);
int extraNum();

int main(void) {

    int amount;

    printf("Generate random contacts. Choose amount: ");
    scanf("%d", &amount);

    for(int i = 0; i < amount; i++) {
        printf("%d%d%d%d%d%d%d%d%d%d, ", trunkPrefix, operators(rand()%7),
        extraNum(), extraNum(), extraNum(), extraNum(),
        extraNum(), extraNum(), extraNum(), extraNum());
    }

}

int operators(int array) {

    int operators[7] = {13, 14, 15, 16, 17, 18, 19};
    return operators[array];

}

int extraNum() {
    return rand()%9;
}
